 $scope.signUp = function(){

    var registrationData = {
      displayName:$scope.name,
      email: $scope.email,
      password: $scope.password
    };

    $http.post("https://apiapp.stamplayapp.com/api/user/v1/users", registrationData);
    });
  };

    }, 
    function(err) {
      console.error(err);
    });
  };
